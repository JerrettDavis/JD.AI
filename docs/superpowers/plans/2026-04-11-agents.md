# Agents Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade the Agents page to a split-panel detail view with agent toolbar and Overview/Tools/Skills tabs (#473), and add a new Skills page at `/agents/skills` with status filtering, per-skill toggle, and configure modal (#474).

**Architecture:** Task A extends the existing `Agents.razor` by introducing a selected-agent state, a side detail panel rendered as a `MudDrawer` (persistent, right-anchor), and three tab views backed by new `AgentDetailInfo` and `SkillInfo` models fetched from new `GatewayApiClient` methods. Task B creates `Skills.razor` as a standalone page under the `/agents/skills` route using `SkillInfo` model cards grouped by category with a configure `MudDialog`. Both tasks add new API client methods, new models, bunit tests, and Playwright/Reqnroll feature files.

**Tech Stack:** Blazor WASM .NET 10, MudBlazor v8.5.1, xunit/bunit, Playwright/Reqnroll

public async Task<AgentDetailInfo?> GetAgentDetailAsync(string id)

    private AgentInfo[]? _agents;
    private bool _loading = true;
    private string? _selectedId;
    private EventHandler<ActivityEventArgs>? _activityHandler;

    protected override async Task OnInitializedAsync()
    {
        _activityHandler = (_, e) =>
        {
            var type = e.Activity.EventType;
            if (type.Contains("agent", StringComparison.OrdinalIgnoreCase) ||
                type.Contains("spawn", StringComparison.OrdinalIgnoreCase))
            {
                InvokeAsync(async () =>
                {
                    Snackbar.Add($"Agent update: {e.Activity.Message}", Severity.Info);
                    await LoadAsync();
                    StateHasChanged();
                });
            }
        };
        SignalR.OnActivityEvent += _activityHandler;
        await LoadAsync();
    }

    private async Task LoadAsync()
    {
        _loading = true;
        try { _agents = await Api.GetAgentsAsync(); }
        catch { _agents = null; }
        _loading = false;
    }

    private async Task SelectAgentAsync(string id)
    {
        _selectedId = id;
        StateHasChanged();
        await Task.CompletedTask;
    }

    private async Task CopyIdAsync()
    {
        if (_selectedId is null) return;
        await JS.InvokeVoidAsync("navigator.clipboard.writeText", _selectedId);
        Snackbar.Add("Agent ID copied", Severity.Info);
    }

    private async Task SetDefaultAsync()
    {
        if (_selectedId is null) return;
        try
        {
            await Api.SetDefaultAgentAsync(_selectedId);
            Snackbar.Add($"Agent '{_selectedId}' set as default", Severity.Success);
        }
        catch (Exception ex) { Snackbar.Add($"Failed: {ex.Message}", Severity.Error); }
    }

    private async Task SpawnAgentAsync()
    {
        var parameters = new DialogParameters<SpawnAgentDialog> { { x => x.Api, Api } };
        var options = new DialogOptions { MaxWidth = MaxWidth.Small, FullWidth = true };
        var dialog = await DialogService.ShowAsync<SpawnAgentDialog>("Spawn Agent", parameters, options);
        var result = await dialog.Result;
        if (result is not null && !result.Canceled)
            await LoadAsync();
    }

    private async Task DeleteAgentAsync(string id)
    {
        var confirm = await DialogService.ShowMessageBox("Stop Agent", $"Stop agent '{id}'?", "Stop", "Cancel");
        if (confirm == true)
        {
            try
            {
                await Api.DeleteAgentAsync(id);
                Snackbar.Add($"Agent '{id}' stopped", Severity.Success);
                if (_selectedId == id) _selectedId = null;
                await LoadAsync();
            }
            catch (Exception ex) { Snackbar.Add($"Failed: {ex.Message}", Severity.Error); }
        }
    }

    public void Dispose()
    {
        if (_activityHandler is not null)
            SignalR.OnActivityEvent -= _activityHandler;
    }
}

            @if (_saving) { <MudProgressCircular Size="Size.Small" Indeterminate="true" Class="mr-2" /> }
            Save
        </MudButton>
    </DialogActions>
</MudDialog>

@code {
    [CascadingParameter] private IMudDialogInstance MudDialog { get; set; } = null!;
    [Parameter, EditorRequired] public SkillInfo Skill { get; set; } = null!;
    [Parameter, EditorRequired] public GatewayApiClient Api { get; set; } = null!;

    private Dictionary<string, string> _config = [];
    private bool _saving;

    protected override void OnInitialized()
    {
        _config = new Dictionary<string, string>(Skill.Config);
    }

    private Color StatusColor => Skill.Status switch
    {
        SkillStatus.Ready => Color.Success,
        SkillStatus.NeedsSetup => Color.Warning,
        _ => Color.Default,
    };

    private void Cancel() => MudDialog.Cancel();

    private async Task SaveAsync()
    {
        _saving = true;
        try
        {
            await Api.UpdateSkillConfigAsync(Skill.Id, _config);
            Snackbar.Add($"Skill '{Skill.Name}' configuration saved", Severity.Success);
            MudDialog.Close(DialogResult.Ok(true));
        }
        catch (Exception ex)
        {
            Snackbar.Add($"Failed: {ex.Message}", Severity.Error);
        }
        finally { _saving = false; }
    }
}
